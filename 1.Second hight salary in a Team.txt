 #include <stdio.h>
 int main() {
  int n,i;
  scanf("%d", &n);
  int arr[n];
  for(i = 0; i < n; i++) {
    scanf("%d", &arr[i]);}
    int largest = arr[0];
    int second = -1;
    for( i = 1; i < n; i++) {
      if(arr[i] > largest) {
        second = largest;
        largest = arr[i]; }
        else if(arr[i] < largest && arr[i] > second) {
          second = arr[i]; }
          }
          printf("%d", second);
          return 0;
}
