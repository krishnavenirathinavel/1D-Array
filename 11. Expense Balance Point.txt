 
 #include <stdio.h>
 int main() {
   int n, arr[100], i;
   int total = 0, leftSum = 0;
   scanf("%d", &n);
   for(i = 0; i < n; i++) {
     scanf("%d", &arr[i]);
     total += arr[i]; }
     for(i = 0; i < n; i++)
     {
       int rightSum = total - leftSum - arr[i];
       if(leftSum == rightSum)
        {
            printf("%d", i);
            return 0;
        }
        leftSum += arr[i];
    }
    printf("-1");

    return 0;
}
