 #include <stdio.h>
 int main() {
   int n, arr[100];
   scanf("%d", &n);
   for(int i = 0; i < n; i++) {
    scanf("%d", &arr[i]);
    }
    int maxLen = 1, currLen = 1;
    for(int i = 1; i < n; i++)
    {
      if(arr[i] > arr[i-1])
      {
            currLen++;
        }
        else {
            currLen = 1;
        }
        if(currLen > maxLen) {
            maxLen = currLen;
        }
    }
    printf("%d", maxLen);
    return 0;
}
