 #include <stdio.h>
 int main() {
   int n, price;
   int min = -1;
   scanf("%d", &n);
   for(int i = 0; i < n; i++) {
     scanf("%d", &price);
     if(price > 0) {          
       if(min == -1 || price < min) {
         min = price;
         }
        }
    }
    if(min == -1) {
      printf("No positive");
    } else {
        printf("%d", min);
    }

    return 0;
}
