 #include <stdio.h>
 int main() {
   int n, arr[100], i, max, temp;
   scanf("%d", &n);
   for(i = 0; i < n; i++) {
     scanf("%d", &arr[i]);
     }
     max = arr[n-1];
     arr[n-1] = -1;
     for(i = n-2; i >= 0; i--) {
       temp = arr[i];
       arr[i] = max;
       if(temp > max) {
         max = temp;
        }
        }
        for(i = 0; i < n; i++) {
          printf("%d ", arr[i]);
          
        }
        return 0;
}
