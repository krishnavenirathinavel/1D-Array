 #include <stdio.h>

int main() {

    int N,i;
    scanf("%d", &N);
    int rolls, sum = 0;
    for(i = 0; i < N-1; i++) {
      scanf("%d", &rolls);
      sum =  sum  + rolls;   }
      int expected_sum = N * (N + 1) / 2;  
      int missing = expected_sum - sum;
      printf("%d\n", missing);
      return 0;
  
}
