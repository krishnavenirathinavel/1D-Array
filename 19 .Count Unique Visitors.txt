 #include <stdio.h>
 int main() {
   int n, arr[100];
   int count = 0;
   scanf("%d", &n);
   for(int i = 0; i < n; i++) {
     scanf("%d", &arr[i]); }
     for(int i = 0; i < n; i++) {
       int found = 0;
       for(int j = 0; j < i; j++) {
         if(arr[i] == arr[j]) {
           found = 1;
           break;
           }
         
       }
       if(found == 0) {
         count++;
        }
    }
    printf("%d", count);
    return 0;
}
